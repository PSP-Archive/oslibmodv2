void oslDrawOsk ( )

Draws the OSK After drawing it you should check if the user closed it. Remember to call oslEndOsk.

if (oslOskIsActive()){
oslDrawOsk();
if (oslGetOskStatus() == PSP_UTILITY_DIALOG_NONE){
//The user closed the OSK
oslEndOsk();
}
}
